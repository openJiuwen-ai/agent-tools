# OpenAPI （插件市场）

## 接口规范文档

```yaml
openapi: 3.1.0
info:
  title: 插件市场 API
  description: 插件上传、发布与下载接口
  version: 1.0.0
servers:
  - url: http://localhost:8100
    description: 本地开发环境
  - url: https://market.example.com
    description: 生产环境
paths:
  /api/v1/plugins:
    post:
      summary: 发布插件
      description: 发布插件到指定空间，同时完成文件上传和发布操作
      operationId: publishPlugin
      tags:
        - 插件管理
      security:
        - bearerAuth: []
      parameters:
        - name: X-Checksum-SHA256
          in: header
          required: true
          description: 插件包文件的 SHA256 校验和（小写十六进制字符串）
          schema:
            type: string
            pattern: "^[a-f0-9]{64}$"
            example: "a1b2c3d4e5f67890a1b2c3d4e5f67890a1b2c3d4e5f67890a1b2c3d4e5f67890"
      requestBody:
        required: true
        content:
          multipart/form-data:
            schema:
              type: object
              required:
                - user_id
                - file
              properties:
                user_id:
                  type: string
                  description: 用户ID
                  example: "63429951"
                file:
                  type: string
                  format: binary
                  description: 插件包文件（.zip格式）
                plugin_id:
                  type: string
                  description: 插件ID，为已存在插件添加新版本时必填。首次发布插件时请勿提供此字段，系统将自动生成。
                  example: "3589119244ed45c29f98038642872858"
                plugin_version:
                  type: string
                  description: 版本号，不填则从 plugin.yaml 读取。格式：主版本号.次版本号.修订号（如 1.0.0）
                  pattern: "^[0-9]+\\.[0-9]+\\.[0-9]+$"
                  example: "1.0.0"
                version_desc:
                  type: string
                  description: 版本说明
                  example: "首次发布，支持天气查询功能"
                force:
                  type: boolean
                  description: 是否强制覆盖同名同版本
                  default: false
            encoding:
              file:
                contentType: application/zip
      responses:
        '200':
          description: 发布成功
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/PluginPublishResponse'
              example:
                code: 200
                data:
                  plugin_id: "3589119244ed45c29f98038642872858"
                  name: "weather-plugin"
                  version: "1.0.0"
                  status: "ACTIVE"
                  published_at: "2025-01-01T00:00:00Z"
                  storage_url: "plugins/xxx/xxx/1.0.0/weather-plugin_1.0.0.zip"
                message: "Publish plugin successfully"
        '400':
          description: 请求参数错误或校验和不匹配
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              examples:
                invalid_file_format:
                  summary: 文件格式错误
                  value:
                    code: 400
                    data: null
                    error: "invalid_file_format"
                    message: "仅支持 .zip 格式的插件包文件"
                checksum_mismatch:
                  summary: 校验和不匹配
                  value:
                    code: 400
                    data: null
                    error: "checksum_mismatch"
                    message: "文件校验和不匹配，文件可能在传输过程中损坏"
        '403':
          description: 权限不足
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              example:
                code: 403
                data: null
                error: "permission_denied"
                message: "您没有权限在该空间发布插件"
        '404':
          description: 插件不存在
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              example:
                code: 404
                data: null
                error: "plugin_not_found"
                message: "插件 '3589119244ed45c29f98038642872858' 不存在，无法添加新版本"
        '409':
          description: 版本冲突
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/VersionConflictResponse'
              examples:
                version_conflict:
                  summary: 版本冲突（应用层检测）
                  value:
                    code: 409
                    data:
                      existing_plugin:
                        plugin_id: "3589119244ed45c29f98038642872858"
                        version: "1.0.0"
                    error: "version_conflict"
                    message: "插件 'weather-plugin' 版本 '1.0.0' 已存在，如需覆盖请设置 force=true"
                version_exists:
                  summary: 版本已存在（数据库约束触发）
                  value:
                    code: 409
                    data:
                      existing_version: "1.0.0"
                    error: "version_exists"
                    message: "插件版本 '1.0.0' 已存在，如需覆盖请设置 force=true"
                plugin_name_exists:
                  summary: 插件名称已存在
                  value:
                    code: 409
                    data: null
                    error: "plugin_name_exists"
                    message: "您已发布过同名插件 'weather-plugin'，请使用其他名称或为现有插件添加新版本"
        '413':
          description: 文件过大
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              example:
                code: 413
                data: null
                error: "file_too_large"
                message: "文件大小超过限制（最大512MB）"
        '422':
          description: 请求验证失败
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              examples:
                manifest_validation_failed:
                  summary: 版本号格式错误
                  value:
                    code: 422
                    data: null
                    error: "manifest_validation_failed"
                    message: "版本号格式错误，必须为 <主版本号>.<次版本号>.<修订号>，例如 1.0.0、1.0.1（不应有 v 前缀）"
                plugin_id_mismatch:
                  summary: plugin_id 与插件包不一致（填写的 plugin_id 与 name 对应的插件不匹配）
                  value:
                    code: 422
                    data:
                      expected_plugin_id: "plugin_xyz789"
                    error: "plugin_id_mismatch"
                    message: "plugin_id 与插件包不匹配：您填写的 plugin_id='3589119244ed45c29f98038642872858' 与插件名称 'weather-plugin' 对应的插件不一致"
                plugin_id_required:
                  summary: 发布新版本时必须填写 plugin_id
                  value:
                    code: 422
                    data:
                      expected_plugin_id: "3589119244ed45c29f98038642872858"
                    error: "plugin_id_required"
                    message: "当前名称下已存在插件，发布新版本时必须填写 plugin_id"
        '500':
          description: 服务器内部错误
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              examples:
                internal_server_error:
                  summary: 服务器内部错误
                  value:
                    code: 500
                    data: null
                    error: "internal_server_error"
                    message: "服务器内部错误，请稍后重试"
                storage_error:
                  summary: 存储上传失败
                  value:
                    code: 500
                    data: null
                    error: "storage_error"
                    message: "插件包上传失败"
  /api/v1/artifacts/{id}:
    get:
      summary: 获取插件下载链接
      description: 根据市场资产ID获取下载链接，返回最新版本的下载地址
      operationId: downloadPlugin
      tags:
        - 插件管理
      parameters:
        - name: id
          in: path
          required: true
          description: 市场资产ID
          schema:
            type: string
            example: "11112222333344445555666677778888"
      responses:
        '200':
          description: 获取下载链接成功
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/DownloadResponse'
              example:
                code: 200
                data:
                  download_url: "https://xxx/plugins/xxx/11112222333344445555666677778888/1.0.0/weather-plugin_1.0.0.zip"
                  asset_id: "11112222333344445555666677778888"
                  name: "weather-plugin"
                  version: "1.0.0"
                  file_size: 102400
                  checksum_sha256: "a1b2c3d4e5f6789012345678901234567890abcdef1234567890abcdef123456"
                message: "ok"
        '404':
          description: 插件不存在
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              example:
                code: 404
                data: null
                error: "plugin_not_found"
                message: "插件 '11112222333344445555666677778888' 不存在"
        '500':
          description: 服务器内部错误
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              examples:
                storage_error:
                  summary: 存储读取失败
                  value:
                    code: 500
                    data: null
                    error: "storage_error"
                    message: "读取插件包元数据失败"
                db_error:
                  summary: 下载统计更新失败
                  value:
                    code: 500
                    data: null
                    error: "db_error"
                    message: "更新下载统计失败"
components:
  schemas:
    DownloadResponse:
      type: object
      required:
        - code
        - data
        - message
      properties:
        code:
          type: integer
          description: 状态码
          example: 200
        data:
          type: object
          required:
            - download_url
            - asset_id
            - name
            - version
            - file_size
            - checksum_sha256
          properties:
            download_url:
              type: string
              description: 下载链接
            asset_id:
              type: string
              description: 插件ID
            name:
              type: string
              description: 插件名称
            version:
              type: string
              description: 版本号
            file_size:
              type: integer
              description: 文件大小（字节）
            checksum_sha256:
              type: string
              description: 文件SHA256校验和
        message:
          type: string
          description: 响应消息
    PluginPublishResponse:
      type: object
      required:
        - code
        - data
        - message
      properties:
        code:
          type: integer
          description: 状态码
          example: 200
        data:
          type: object
          required:
            - plugin_id
            - name
            - version
            - status
            - published_at
            - storage_url
          properties:
            plugin_id:
              type: string
              description: 插件ID
            name:
              type: string
              description: 插件名称
            version:
              type: string
              description: 版本号
            status:
              type: string
              enum: [pending_review, approved, rejected, deprecated]
              description: 审核状态
            published_at:
              type: string
              format: date-time
              description: 发布时间
            storage_url:
              type: string
              description: 插件包存储URL
        message:
          type: string
          description: 响应消息
    ErrorResponse:
      type: object
      required:
        - code
        - data
        - error
        - message
      properties:
        code:
          type: integer
          description: 状态码
        data:
          type: object
          nullable: true
          description: 响应数据
        error:
          type: string
          description: 错误代码
        message:
          type: string
          description: 错误消息
    VersionConflictResponse:
      type: object
      required:
        - code
        - data
        - error
        - message
      properties:
        code:
          type: integer
          description: 状态码
          example: 409
        data:
          type: object
          required:
            - existing_plugin
          properties:
            existing_plugin:
              type: object
              required:
                - plugin_id
                - version
              properties:
                plugin_id:
                  type: string
                  description: 已存在的插件ID
                version:
                  type: string
                  description: 已存在的版本号
        error:
          type: string
          description: 错误代码
          example: "version_conflict"
        message:
          type: string
          description: 错误消息
  securitySchemes:
    bearerAuth:
      type: http
      scheme: bearer
      bearerFormat: JWT
      description: 使用 JWT Bearer Token 进行认证
```
